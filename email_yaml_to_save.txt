reporting:
   live_dashboard:
       features:
           - real_time_status_view (via SSE)

   executive_reports:
       time_periods: configurable (shift, day, week, month, custom)
       definitions:    # explicit formulas (best-guess standard manufacturing practice)
           - mttr:
                   description: Mean Time To Repair
                   formula: Total repair time (sum of WorkLog.time_spent_minutes for resolved events) / Number of resolved repairs (critical + non-critical configurable)
                   note: Excludes wait time unless explicitly logged; filterable by severity
           - mtbf:
                   description: Mean Time Between Failures
                   formula: Total operational uptime (derived from resolved_at - previous resolved_at or running periods) / Number of failures (downtime events marked down)
                   note: Uses StatusHistory for precise uptime calculation; excludes planned downtime if tagged later
           - downtime_by_equipment:
                   description: Top offending equipment (total downtime minutes)
           - downtime_by_module:
                   description: Component failure patterns
           - technician_performance:
                   description: Workload (events handled), average resolution time, parts cost
           - peak_failure_times:
                   description: Shift / hour pattern analysis
           - repair_cost_summary:
                   description: Financial impact (parts + optional labor rate)
       export_formats: CSV, Excel, PDF (with charts via backend or frontend)
       scheduling: future (phase 3+)
       visibility: role-based (costs hidden from operators by default)
